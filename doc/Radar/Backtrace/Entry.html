def initialize(line)
  # Regex pulled from HoptoadNotifier. Thanks!
  _, file, line, method = line.match(/^([^:]+):(\d+)(?::in `([^']+)')?$/).to_a
  self[:file] = file
  self[:line] = line
  self[:method] = method
end
