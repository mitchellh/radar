def extract_http_headers(env)
  env.inject({}) do |acc, data|
    k, v = data

    if k =~ /^HTTP_(.+)$/
      # Convert things like HTTP_CONTENT_TYPE to Content-Type (standard
      # HTTP header style)
      k = $1.to_s.split("_").map { |c| c.capitalize }.join("-")
      acc[k] = v
    end

    acc
  end
end
